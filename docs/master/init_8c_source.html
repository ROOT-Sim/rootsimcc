105 static void parse_opt(int key, const char *arg)
106 {
107  bool parse_err = false;
108  switch (key) {
109 
110  case OPT_NPRC:
111  n_lps = parse_ullong_limits(arg, 1, UINT_MAX, &parse_err);
112  break;
113 
114  case OPT_LOG:
115  log_level = parse_ullong_limits(arg, 0, 6, &parse_err);
116  break;
117 
118  case OPT_SIMT:
119  global_config.termination_time = parse_ldouble_limits(arg, 0,
120  SIMTIME_MAX, &parse_err);
121  break;
122 
123  case OPT_GVT:
124  global_config.gvt_period = parse_ullong_limits(arg, 1, 10000,
125  &parse_err) * 1000;
126  break;
127 
128  case OPT_NP:
129  n_threads = parse_ullong_limits(arg, 1, UINT_MAX, &parse_err);
130  break;
131 
132  case OPT_BIND:
133  global_config.core_binding = false;
134  break;
135 
136  case OPT_SERIAL:
137  global_config.is_serial = true;
138  break;
139 
140  case OPT_SEED:
141  global_config.prng_seed = parse_ullong_limits(arg, 0,
142  UINT64_MAX, &parse_err);
143  break;
144 
145  case AP_KEY_INIT:
146  n_lps = 0;
147  n_threads = 0;
148  global_config.termination_time = SIMTIME_MAX;
149  global_config.gvt_period = 200000;
150  global_config.verbosity = 0;
151  global_config.is_serial = false;
152  global_config.core_binding = true;
153  log_colored = io_terminal_can_colorize();
154  break;
155 
156  case AP_KEY_FINI:
157  // if the threads count has not been supplied, the other checks
158  // are superfluous: the serial runtime simply will ignore the
159  // field while the parallel one will use the set count of cores
160  // (spitting a warning if it must use less cores than available)
161  if (n_threads == 0) {
162  n_threads = global_config.is_serial ? 1 :
163  thread_cores_count();
164  } else if (global_config.is_serial) {
165  arg_parse_error("requested a serial simulation with %u threads",
166  n_threads);
167  } else if (n_threads > n_lps) {
168  arg_parse_error("requested a simulation with %u threads and %" PRIu64 " LPs",
169  n_threads, n_lps);
170  } else if(n_threads > thread_cores_count()) {
171  arg_parse_error("demanding %u cores, which are more than available (%u)",
172  n_threads, thread_cores_count());
173  }
174 
175  if(n_lps == 0)
176  arg_parse_error("number of LPs was not provided \"--lp\"");
177 
178  log_logo_print();
179  print_config();
180  }
181 
182  if (unlikely(parse_err)) {
183  size_t i;
184  for (i = 0; ap_options[i].key != key; ++i);
185  arg_parse_error("invalid value \"%s\" in the %s option.", arg,
186  ap_options[i].name);
187  }
188 }
